Insertion Sort
        var insertionSort = function(arr){
          for(var j = 1; j < arr.length; j++){
            var key = arr[j],
                i = j - 1;

             while(i != 0 & arr[i] > key){
              arr[i+1] = arr[i];
              i--;
             }
             arr[i+1] = key;
          }

          return arr;
        }
